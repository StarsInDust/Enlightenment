~2 obj_hero_fight~

By StarsInDust

5/11/2026

 

//~~~~~~~~~~~~~~  obj_hero_fight - CREATE EVENT  ~~~~~~~~~~~~~~~~~~~

 

// Sprite References

spr_hero_idle = spr_hero_idle_128;

spr_hero_attack = spr_hero_attack_128;

spr_hero_hit = spr_hero_hit_128;

spr_hero_death = spr_hero_death_128;

 

// Animation Settings

image_speed = 0;

image_index = 0;

 

// State Machine

enum HERO_STATE {

    IDLE,

    ATTACK,

    LUNGING,

    RETURNING,

    HIT,

    DEATH,

    DEAD

}

 

current_state = HERO_STATE.IDLE;

sprite_index = spr_hero_idle;

 

// Combat Stats

if (variable_global_exists("hero_max_hp") && global.hero_max_hp != undefined) {

    max_hp = global.hero_max_hp;

    current_hp = global.hero_current_hp;

} else {

    max_hp = 100;

    current_hp = max_hp;

}

 

attack_damage = 20;

is_player_turn = true;

 

// Lunge System

home_x = room_width * 0.35;

home_y = room_height * 0.5;

lunge_target_x = home_x;

lunge_target_y = home_y;

lunge_speed = 12;

return_speed = 8;

lunge_stop_distance = 170;

 

// Knockback

knockback_x = 0;

knockback_friction = 0.8;

 

// Animation Tracking

animation_complete = false;

attack_has_dealt_damage = false;

 

// Visual Effects

hit_flash_timer = 0;

image_xscale = 1;

image_yscale = 1;

is_dead = false;

 

// Position

x = home_x;

y = home_y;

 

 

 

// ~~~~~~~~~~~~~~~~    obj_hero_fight - STEP EVENT   ~~~~~~~~~~~~~~

 

// Animation Completion Check

if (floor(image_index) >= image_number - 1) {

    animation_complete = true;

} else {

    animation_complete = false;

}

 

// Update Timers

if (hit_flash_timer > 0) hit_flash_timer--;

 

// Apply Knockback

if (abs(knockback_x) > 0.5) {

    x += knockback_x;

    knockback_x *= knockback_friction;

} else {

    knockback_x = 0;

}

 

// State Machine

switch (current_state) {

 

    case HERO_STATE.IDLE:

        if (sprite_index != spr_hero_idle) {

            sprite_index = spr_hero_idle;

            image_index = 0;

            image_speed = 0;

        }

       

        if (abs(x - home_x) > 2) {

            x = lerp(x, home_x, 0.1);

        }

       

        if (is_player_turn && !is_dead) {

            if (keyboard_check_pressed(vk_space) || mouse_check_button_pressed(mb_left)) {

                if (instance_exists(obj_enemy_fight)) {

                    var enemy_home = obj_enemy_fight.home_x;

                    var enemy_y = obj_enemy_fight.home_y;

                    lunge_target_x = enemy_home - lunge_stop_distance;

                    lunge_target_y = enemy_y;

                }

                current_state = HERO_STATE.LUNGING;

                attack_has_dealt_damage = false;

                is_player_turn = false;

            }

        }

        break;

   

    case HERO_STATE.LUNGING:

        var dist_to_target = point_distance(x, y, lunge_target_x, lunge_target_y);

       

        if (dist_to_target > lunge_speed) {

            var move_dir = point_direction(x, y, lunge_target_x, lunge_target_y);

            x += lengthdir_x(lunge_speed, move_dir);

            y += lengthdir_y(lunge_speed, move_dir);

        } else {

            x = lunge_target_x;

            y = lunge_target_y;

            current_state = HERO_STATE.ATTACK;

            sprite_index = spr_hero_attack;

            image_index = 0;

            image_speed = 0.2;

        }

        break;

   

    case HERO_STATE.ATTACK:

        if (sprite_index != spr_hero_attack) {

            sprite_index = spr_hero_attack;

            image_index = 0;

            image_speed = 0.2;

        }

       

        if (floor(image_index) >= 2 && !attack_has_dealt_damage) {

            attack_has_dealt_damage = true;

           

            if (instance_exists(obj_enemy_fight)) {

                with (obj_enemy_fight) {

                    scr_take_damage_enemy(other.attack_damage);

                    knockback_x = 15;

                }

               

                var fx_x = obj_enemy_fight.x - 40;

                var fx_y = obj_enemy_fight.y;

               

                if (object_exists(obj_hit_effect)) {

                    instance_create_depth(fx_x, fx_y, -200, obj_hit_effect);

                }

            }

        }

       

        if (animation_complete) {

            current_state = HERO_STATE.RETURNING;

        }

        break;

   

    case HERO_STATE.RETURNING:

        var dist_to_home = point_distance(x, y, home_x, home_y);

       

        if (dist_to_home > return_speed) {

            var move_dir = point_direction(x, y, home_x, home_y);

            x += lengthdir_x(return_speed, move_dir);

            y += lengthdir_y(return_speed, move_dir);

        } else {

            x = home_x;

            y = home_y;

            current_state = HERO_STATE.IDLE;

            sprite_index = spr_hero_idle;

            image_index = 0;

            image_speed = 0;

            is_player_turn = true;

        }

        break;

   

    case HERO_STATE.HIT:

        if (sprite_index != spr_hero_hit) {

            sprite_index = spr_hero_hit;

            image_index = 0;

            image_speed = 0.15;

            hit_flash_timer = 10;

        }

       

        if (animation_complete) {

            if (current_hp <= 0) {

                current_state = HERO_STATE.DEATH;

                if (sprite_exists(spr_hero_death)) {

                    sprite_index = spr_hero_death;

                }

                image_index = 0;

            } else {

                current_state = HERO_STATE.IDLE;

                sprite_index = spr_hero_idle;

                image_index = 0;

                image_speed = 0;

                is_player_turn = true;

            }

        }

        break;

   

    case HERO_STATE.DEATH:

        if (sprite_index != spr_hero_death && sprite_exists(spr_hero_death)) {

            sprite_index = spr_hero_death;

            image_index = 0;

            image_speed = 0.1;

        }

       

        if (animation_complete) {

            current_state = HERO_STATE.DEAD;

            is_dead = true;

            image_speed = 0;

           

            if (instance_exists(obj_fight_controller)) {

                obj_fight_controller.fight_lost = true;

            }

        }

        break;

   

    case HERO_STATE.DEAD:

        image_speed = 0;

        break;

}
