~obj_hit_effect~

By StarsInDust

5/11/2026

 

// ~~~~~~~~~~~~   obj_hit_effect - CREATE EVENT  ~~~~~~~~~~~~~~~~~~~

 

lifetime = 15;

max_lifetime = 15;

 

effect_alpha = 1.0;

effect_scale = 1.0;

effect_color = c_white;

 

expand_speed = 0.15;

float_speed = -1;

 

image_angle = irandom(360);

effect_scale = 0.8 + random(0.4);

 

 

 

//  ~~~~~~~~~~~~~~ obj_hit_effect - STEP EVENT ~~~~~~~~~~~~~~~~~~~

 

lifetime -= 1;

 

var progress = lifetime / max_lifetime;

effect_alpha = progress;

 

effect_scale += expand_speed;

 

y += float_speed;

 

if (lifetime <= 0) {

    instance_destroy();

}

 

 

// ~~~~~~~~~~~~~~~~  obj_hit_effect - DRAW EVENT  ~~~~~~~~~~~~~

 

if (effect_alpha > 0) {

   

    draw_set_alpha(effect_alpha * 0.5);

    draw_set_color(c_white);

    draw_circle(x, y, 25 * effect_scale, false);

   

    draw_set_alpha(effect_alpha);

    var slash_len = 35 * effect_scale;

   

    draw_set_color(c_white);

    draw_line_width(

        x - lengthdir_x(slash_len, image_angle),

        y - lengthdir_y(slash_len, image_angle),

        x + lengthdir_x(slash_len, image_angle),

        y + lengthdir_y(slash_len, image_angle),

        4

    );

   

    draw_line_width(

        x - lengthdir_x(slash_len * 0.6, image_angle + 90),

        y - lengthdir_y(slash_len * 0.6, image_angle + 90),

        x + lengthdir_x(slash_len * 0.6, image_angle + 90),

        y + lengthdir_y(slash_len * 0.6, image_angle + 90),

        3

    );

   

    draw_set_color(c_yellow);

    draw_circle(x, y, 10 * effect_scale, false);

   

    draw_set_alpha(1);

    draw_set_color(c_white);

}
