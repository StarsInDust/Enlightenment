// Call this AFTER the hero is spawned in the maze to save respawn point

 

function save_hero_start_position() {

    var hero = instance_find(object_hero, 0);

   

    if (hero != noone) {

        global.hero_start_x = hero.x;

        global.hero_start_y = hero.y;

       

        show_debug_message("=== HERO START POSITION SAVED ===");

        show_debug_message("Position: (" + string(global.hero_start_x) + ", " + string(global.hero_start_y) + ")");

    } else {

        show_debug_message("ERROR: Could not find hero to save starting position!");

    }

}

 

 

// ~~~~~~~~~~~~~~~~FUNCTION: Reset Hero to Start Position ~~~~~~~~~~~~~~~

// Call this when hero loses all lives on current floor

 

function reset_hero_to_start() {

    var hero = instance_find(object_hero, 0);

   

    if (hero != noone) {

        hero.x = global.hero_start_x;

        hero.y = global.hero_start_y;

       

        // Reset hearts to starting amount

        global.hearts = 3;

        if (variable_global_exists("max_energy")) {

            global.energy = global.max_energy;

        }

        if (variable_global_exists("energy_move_accumulator")) {

            global.energy_move_accumulator = 0;

        }

       

        show_debug_message("=== HERO RESET TO START ===");

        show_debug_message("Teleported to: (" + string(global.hero_start_x) + ", " + string(global.hero_start_y) + ")");

        show_debug_message("Hearts restored to: " + string(global.hearts));

        if (variable_global_exists("energy") && variable_global_exists("max_energy")) {

            show_debug_message("Energy restored to: " + string(global.energy) + "/" + string(global.max_energy));

        }

    } else {

        show_debug_message("ERROR: Could not find hero to reset!");

    }

}

 

 

// ~~~~~~~~~~~~~~~~~FUNCTION: Take Damage ~~~~~~~~~~~~~~~~~~~~~~~~~

// Call when hero takes damage (bomb, trap, enemy, etc.)

 

function hero_take_damage(_damage_amount) {

    global.hearts -= _damage_amount;

   

    show_debug_message("Hero took " + string(_damage_amount) + " damage! Hearts: " + string(global.hearts));

   

    // Check if dead

    if (global.hearts <= 0) {

        show_debug_message("Hero died! Resetting to floor start...");

        reset_hero_to_start();

        return true;  // Return true to indicate death occurred

    }

   

    return false;  // Still alive

}

 

 

// ~~~~~~~~~~~~~~~~~~~~~~ FUNCTION: Heal Hero~~~~~~~~~~~~~~

// Call when hero collects a heart

 

function hero_heal(_heal_amount) {

    global.hearts += _heal_amount;

   

    // Cap at maximum

    if (global.hearts > global.max_hearts) {

        global.hearts = global.max_hearts;

    }

   

    show_debug_message("Hero healed " + string(_heal_amount) + "! Hearts: " + string(global.hearts) + "/" + string(global.max_hearts));

}

// ~~~~~~~~~~~~~~FUNCTION: Use Apple for Energy ~~~~~~~~~~~~~~~~~~~~~

// Call when hero manually consumes an apple

 

function hero_use_apple_for_energy() {

    if (!variable_global_exists("energy") || !variable_global_exists("max_energy")) {

        return false;

    }

 

    if (global.apples <= 0) {

        show_debug_message("No apples left to eat.");

        return false;

    }

 

    if (global.energy >= global.max_energy) {

        show_debug_message("Energy already full.");

        return false;

    }

 

    global.apples -= 1;

 

    var restore_amount = 10;

    if (variable_global_exists("energy_restore_per_apple")) {

        restore_amount = global.energy_restore_per_apple;

    }

 

    global.energy += restore_amount;

    if (global.energy > global.max_energy) {

        global.energy = global.max_energy;

    }

 

    show_debug_message("Ate apple. Energy: " + string(global.energy) + "/" + string(global.max_energy) + " | Apples left: " + string(global.apples));

    return true;

}

 

 

// ~~~~~~~~~~~~~~FUNCTION: Drain Energy from Movement ~~~~~~~~~~~~

// _distance_moved should be pixel distance moved this step

 

function hero_drain_energy_by_movement(_distance_moved) {

    if (!variable_global_exists("energy") || !variable_global_exists("max_energy")) {

        return;

    }

 

    if (_distance_moved <= 0) {

        return;

    }

 

    if (!variable_global_exists("energy_move_accumulator")) {

        global.energy_move_accumulator = 0;

    }

 

    var pixels_per_point = 96;

    if (variable_global_exists("energy_pixels_per_point")) {

        pixels_per_point = max(1, global.energy_pixels_per_point);

    }

 

    global.energy_move_accumulator += _distance_moved;

 

    while (global.energy_move_accumulator >= pixels_per_point) {

        global.energy_move_accumulator -= pixels_per_point;

        global.energy -= 1;

 

        if (global.energy <= 0) {

            global.energy = 0;

            break;

        }

    }

}
