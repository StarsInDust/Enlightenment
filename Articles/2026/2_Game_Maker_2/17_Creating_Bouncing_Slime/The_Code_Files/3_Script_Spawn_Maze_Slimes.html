~3 Script Spawn Maze Slimes~

By StarsInDust

5/4/2026

 

// ~~~~~~~~~    SPAWN MAZE SLIMES - Standalone Script ~~~~~~~~~~~~~~~~~~

// Create a script in GameMaker called "spawn_maze_slimes"

// Spawns a set number of slimes in open maze cells

 

function spawn_maze_slimes(_mazeStruct, _cellSize, _objectType, _count) {

    show_debug_message("=== SPAWNING MAZE SLIMES ===");

    show_debug_message("Count: " + string(_count));

 

    var m = _mazeStruct;

    var cell_size = _cellSize;

 

    var spawn_positions = ds_list_create();

 

    for (var i = 1; i < m.w - 1; i++) {

        for (var j = 1; j < m.h - 1; j++) {

            var pos_x = i * cell_size + (cell_size / 2);

            var pos_y = j * cell_size + (cell_size / 2);

 

            var hero = instance_find(object_hero, 0);

            var door = instance_find(obj_door, 0);

 

            var too_close = false;

            var min_distance = 180;

 

            if (hero != noone && point_distance(pos_x, pos_y, hero.x, hero.y) < min_distance) {

                too_close = true;

            }

            if (door != noone && point_distance(pos_x, pos_y, door.x, door.y) < min_distance) {

                too_close = true;

            }

 

            if (!too_close && !position_meeting(pos_x, pos_y, obj_wall_plain) &&

                !position_meeting(pos_x, pos_y, obj_wall_crackedA) &&

                !position_meeting(pos_x, pos_y, obj_wall_crackedB)) {

                ds_list_add(spawn_positions, [pos_x, pos_y, i, j]);

            }

        }

    }

 

    show_debug_message("Valid slime spawn positions: " + string(ds_list_size(spawn_positions)));

 

    ds_list_shuffle(spawn_positions);

 

    var placed = 0;

    var max_place = min(_count, ds_list_size(spawn_positions));

 

    for (var s = 0; s < max_place; s++) {

        var pos = spawn_positions[| s];

        instance_create_layer(pos[0], pos[1], "Instances", _objectType);

        show_debug_message("Placed slime at cell [" + string(pos[2]) + ", " + string(pos[3]) + "]");

        placed++;

    }

 

    ds_list_destroy(spawn_positions);

 

    show_debug_message("Total slimes placed: " + string(placed));

    show_debug_message("=== MAZE SLIME SPAWNING COMPLETE ===");

}
