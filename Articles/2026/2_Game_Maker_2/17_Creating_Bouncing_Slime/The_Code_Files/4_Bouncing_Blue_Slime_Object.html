~4 Bouncing Blue Slime Object~

By StarsInDust

5/4/2026

 

//~~~~~~~~~~~~~~~~~~~~~~~~CREATE EVENT  ~~~~~~~~~~~~~~~~~~~~~~~~

// ~~~~~~~~ obj_blue_bouncing_directional_slime -  CREATE EVENT ~~~~~~~~~~

 

// obj_blue_bouncing_directional_slime - CREATE EVENT

// Small maze slime setup (directional bounce)

 

move_speed = 2;

lunge_speed = 14;

lunge_distance = 192;

lunge_align_tolerance = 48;

lunge_align_speed = 2;

lunge_time_max = 20;

lunge_timer = 0;

lunge_windup_max = 12;

lunge_windup = 0;

lunge_pending = false;

lunge_dir = 0;

lunge_target_x = 0;

lunge_target_y = 0;

lunge_cooldown_max = 60;

lunge_cooldown = 0;

move_dir = choose(0, 90, 180, 270);

 

sprite_up = spr_blue_bouncing_slime_up;

sprite_down = spr_blue_bouncing_slime_down;

sprite_left = spr_blue_bouncing_slime_left;

sprite_right = spr_blue_bouncing_slime_right;

 

sprite_index = sprite_down;

image_speed = 0.2;

 

image_xscale = 0.5;

image_yscale = 0.5;

 

 

//~~~~~~~~~~~~~~~~~~~~~~~~STEP EVENT  ~~~~~~~~~~~~~~~~~~~~~~~~

 

// ~~ obj_blue_bouncing_directional_slime - STEP EVENT  ~~

// Bounce through maze hallways and switch sprites by direction

 

var hero = instance_find(object_hero, 0);

 

if (lunge_cooldown > 0) lunge_cooldown--;

if (lunge_timer > 0) lunge_timer--;

if (lunge_windup > 0) {

    lunge_windup--;

    if (lunge_windup == 0 && lunge_pending) {

        move_dir = lunge_dir;

        lunge_timer = lunge_time_max;

        lunge_cooldown = lunge_cooldown_max;

        lunge_pending = false;

    }

}

 

if (lunge_windup > 0 && lunge_pending) {

    if (lunge_dir == 90 || lunge_dir == 270) {

        var align_dx = clamp(lunge_target_x - x, -lunge_align_speed, lunge_align_speed);

        if (!place_meeting(x + align_dx, y, obj_wall_plain) &&

            !place_meeting(x + align_dx, y, obj_wall_crackedA) &&

            !place_meeting(x + align_dx, y, obj_wall_crackedB)) {

            x += align_dx;

        }

    } else {

        var align_dy = clamp(lunge_target_y - y, -lunge_align_speed, lunge_align_speed);

        if (!place_meeting(x, y + align_dy, obj_wall_plain) &&

            !place_meeting(x, y + align_dy, obj_wall_crackedA) &&

            !place_meeting(x, y + align_dy, obj_wall_crackedB)) {

            y += align_dy;

        }

    }

}

 

if (hero != noone && lunge_timer <= 0 && lunge_cooldown <= 0 && lunge_windup <= 0 && !lunge_pending) {

    var dist_to_hero = point_distance(x, y, hero.x, hero.y);

    if (dist_to_hero <= lunge_distance) {

        var blocked_line = collision_line(x, y, hero.x, hero.y, obj_wall_plain, false, true) ||

                           collision_line(x, y, hero.x, hero.y, obj_wall_crackedA, false, true) ||

                           collision_line(x, y, hero.x, hero.y, obj_wall_crackedB, false, true);

        if (!blocked_line) {

            var dx = hero.x - x;

            var dy = hero.y - y;

            var aligned_h = abs(dy) <= lunge_align_tolerance;

            var aligned_v = abs(dx) <= lunge_align_tolerance;

 

            if (aligned_h || aligned_v) {

                if (aligned_h && !aligned_v) {

                    lunge_dir = (dx > 0) ? 0 : 180;

                } else if (aligned_v && !aligned_h) {

                    lunge_dir = (dy > 0) ? 270 : 90;

                } else {

                    lunge_dir = (abs(dx) > abs(dy)) ? ((dx > 0) ? 0 : 180) : ((dy > 0) ? 270 : 90);

                }

                lunge_target_x = hero.x;

                lunge_target_y = hero.y;

                lunge_windup = lunge_windup_max;

                lunge_pending = true;

            } else {

                var align_dir;

                if (abs(dx) > abs(dy)) {

                    align_dir = (dy > 0) ? 270 : 90;

                } else {

                    align_dir = (dx > 0) ? 0 : 180;

                }

 

                var align_blocked = false;

                if (align_dir == 0) {

                    align_blocked = place_meeting(x + move_speed, y, obj_wall_plain) ||

                                    place_meeting(x + move_speed, y, obj_wall_crackedA) ||

                                    place_meeting(x + move_speed, y, obj_wall_crackedB);

                } else if (align_dir == 180) {

                    align_blocked = place_meeting(x - move_speed, y, obj_wall_plain) ||

                                    place_meeting(x - move_speed, y, obj_wall_crackedA) ||

                                    place_meeting(x - move_speed, y, obj_wall_crackedB);

                } else if (align_dir == 90) {

                    align_blocked = place_meeting(x, y - move_speed, obj_wall_plain) ||

                                    place_meeting(x, y - move_speed, obj_wall_crackedA) ||

                                    place_meeting(x, y - move_speed, obj_wall_crackedB);

                } else if (align_dir == 270) {

                    align_blocked = place_meeting(x, y + move_speed, obj_wall_plain) ||

                                    place_meeting(x, y + move_speed, obj_wall_crackedA) ||

                                    place_meeting(x, y + move_speed, obj_wall_crackedB);

                }

 

                if (!align_blocked) {

                    move_dir = align_dir;

                }

            }

        }

    }

}

 

var current_speed = (lunge_windup > 0) ? 0 : ((lunge_timer > 0) ? lunge_speed : move_speed);

 

var hsp = 0;

var vsp = 0;

 

switch (move_dir) {

    case 0:   hsp = current_speed;  break; // right

    case 180: hsp = -current_speed; break; // left

    case 90:  vsp = -current_speed; break; // up

    case 270: vsp = current_speed;  break; // down

}

 

var blocked = place_meeting(x + hsp, y + vsp, obj_wall_plain) ||

              place_meeting(x + hsp, y + vsp, obj_wall_crackedA) ||

              place_meeting(x + hsp, y + vsp, obj_wall_crackedB);

 

if (blocked) {

    var options = [];

 

    var open_right = !(place_meeting(x + current_speed, y, obj_wall_plain) ||

                       place_meeting(x + current_speed, y, obj_wall_crackedA) ||

                       place_meeting(x + current_speed, y, obj_wall_crackedB));

    if (open_right) array_push(options, 0);

 

    var open_left = !(place_meeting(x - current_speed, y, obj_wall_plain) ||

                      place_meeting(x - current_speed, y, obj_wall_crackedA) ||

                      place_meeting(x - current_speed, y, obj_wall_crackedB));

    if (open_left) array_push(options, 180);

 

    var open_up = !(place_meeting(x, y - current_speed, obj_wall_plain) ||

                    place_meeting(x, y - current_speed, obj_wall_crackedA) ||

                    place_meeting(x, y - current_speed, obj_wall_crackedB));

    if (open_up) array_push(options, 90);

 

    var open_down = !(place_meeting(x, y + current_speed, obj_wall_plain) ||

                      place_meeting(x, y + current_speed, obj_wall_crackedA) ||

                      place_meeting(x, y + current_speed, obj_wall_crackedB));

    if (open_down) array_push(options, 270);

 

    if (array_length(options) > 0) {

        move_dir = options[irandom(array_length(options) - 1)];

    } else {

        move_dir = (move_dir + 180) mod 360;

    }

} else {

    x += hsp;

    y += vsp;

}

 

if (move_dir == 0) {

    sprite_index = sprite_right;

} else if (move_dir == 180) {

    sprite_index = sprite_left;

} else if (move_dir == 90) {

    sprite_index = sprite_up;

} else if (move_dir == 270) {

    sprite_index = sprite_down;

}
