// obj_environment_controller - CREATE EVENT

// Place one instance of this in any environment room (town, cave, forest, shop, etc.)

 

// Spawn Hero

if (!instance_exists(object_hero)) {

    var spawn_x = room_width / 2;

    var spawn_y = room_height - 150;

   

    var hero = instance_create_depth(spawn_x, spawn_y, -50, object_hero);

    hero.image_xscale = 3;

    hero.image_yscale = 3;

    hero.sprite_index = sprite_hero_back;

} else {

    var hero = instance_find(object_hero, 0);

    hero.x = room_width / 2;

    hero.y = room_height - 150;

    hero.image_xscale = 3;

    hero.image_yscale = 3;

    hero.sprite_index = sprite_hero_back;

}
